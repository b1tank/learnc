> Generalise the sorter so it can sort on **fields**: a `+f` option selects field `f` (whitespace-separated) as the sort key, and combinations `+f1 -f2 +f3` mean "primary key field 1, secondary field 3 (descending takes the comparator slot in actual `sort(1)`)" - adapt to your taste.

## Approach

A field selector tells the comparator "advance to the f-th whitespace-separated token, then compare from there". For multi-key sort, run the comparators in order and return the first non-zero result.

```c:starter
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

/* return pointer to start of field n (0-based) within s, or NULL */
static const char *field_start(const char *s, int n) {
    int field = 0;
    while (*s) {
        while (*s && isspace((unsigned char)*s)) ++s;
        if (*s == '\0') break;
        if (field == n) return s;
        while (*s && !isspace((unsigned char)*s)) ++s;
        ++field;
    }
    return NULL;
}

/* compare s and t restricted to field n */
static int cmp_field(const char *s, const char *t, int n) {
    const char *ps = field_start(s, n);
    const char *pt = field_start(t, n);
    if (!ps) return pt ? -1 : 0;
    if (!pt) return 1;
    while (*ps && !isspace((unsigned char)*ps) &&
           *pt && !isspace((unsigned char)*pt)) {
        if (*ps != *pt) return (unsigned char)*ps - (unsigned char)*pt;
        ++ps; ++pt;
    }
    /* tie-breaker: end of field on either side */
    int es = (!*ps || isspace((unsigned char)*ps));
    int et = (!*pt || isspace((unsigned char)*pt));
    return es == et ? 0 : (es ? -1 : 1);
}

static int sort_field = 0;

static int cmp_lines(const void *a, const void *b) {
    const char *sa = *(const char *const *)a;
    const char *sb = *(const char *const *)b;
    return cmp_field(sa, sb, sort_field);
}

#define MAXLINES 5000
static char *lines[MAXLINES];

int main(int argc, char *argv[]) {
    for (int i = 1; i < argc; ++i)
        if (argv[i][0] == '+') sort_field = atoi(argv[i] + 1);

    int n = 0;
    char buf[4096];
    while (n < MAXLINES && fgets(buf, sizeof buf, stdin)) lines[n++] = strdup(buf);
    qsort(lines, n, sizeof lines[0], cmp_lines);
    for (int i = 0; i < n; ++i) fputs(lines[i], stdout);
    return 0;
}
```

Usage:

```bash
$ printf 'alice 30\nbob 25\ncarol 27\n' | ./sort +1
bob 25
carol 27
alice 30
```

Sort on field 1 (the number) gives ascending age. Sorting on field 0 (default) is the usual name sort.

## Notes from the author

- "Field" needs a definition: here it's whitespace-separated, 0-indexed. Real `sort(1)` lets you pick the separator (`-t`) and the field range (`-k`).
- For numeric fields, you'd add a comparison that interprets the substring as a number. Mixing alphabetic and numeric keys gets fiddly - that's why `sort -k` has so many flags.
- Multi-key sort is just a chain of `if (rA != 0) return rA;` checks. The pattern generalises to *N* keys without complicating the comparator significantly.
