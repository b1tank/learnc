> Extend the previous exercise: accept `-m +n` as "every `n` columns starting at column `m`". For example `-3 +5` means stops at columns 3, 8, 13, 18, ...

## Approach

Two patterns of argument:

- `<number>` - a fixed tab stop column.
- `-<m> +<n>` - start at `m`, step by `n` forever.

Mix and match: `4 8 -16 +4` → fixed stops at 4 and 8, then every 4 starting at 16.

```c:starter
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAXTABS 128
static int fixed_tabs[MAXTABS];
static int n_fixed = 0;
static int interval_start = -1;     /* -1 = no infinite interval */
static int interval_step  = 8;

/* return next stop strictly > col */
static int next_stop(int col) {
    /* find fixed tab > col */
    int best = -1;
    for (int i = 0; i < n_fixed; ++i)
        if (fixed_tabs[i] > col && (best == -1 || fixed_tabs[i] < best))
            best = fixed_tabs[i];
    /* candidate from interval */
    if (interval_start != -1) {
        int next;
        if (col < interval_start) next = interval_start;
        else                      next = interval_start
                                  + ((col - interval_start) / interval_step + 1) * interval_step;
        if (best == -1 || next < best) best = next;
    }
    /* fallback: default step of 8 from the last fixed/interval anchor */
    if (best == -1)
        best = ((col / 8) + 1) * 8;
    return best;
}

static void detab_stream(void) {
    int c, col = 0;
    while ((c = getchar()) != EOF) {
        if (c == '\t') {
            int target = next_stop(col);
            while (col < target) { putchar(' '); ++col; }
        } else if (c == '\n') {
            putchar('\n'); col = 0;
        } else {
            putchar(c); ++col;
        }
    }
}

int main(int argc, char *argv[]) {
    for (int i = 1; i < argc; ++i) {
        if (argv[i][0] == '-' && i + 1 < argc && argv[i+1][0] == '+') {
            interval_start = atoi(argv[i] + 1);
            interval_step  = atoi(argv[i+1] + 1);
            ++i;
        } else {
            int t = atoi(argv[i]);
            if (n_fixed < MAXTABS) fixed_tabs[n_fixed++] = t;
        }
    }
    detab_stream();
    return 0;
}
```

Usage:

```bash
$ printf 'a\tb\tc\td\n' | ./detab -3 +5
a  b    c    d        (stops at 3, 8, 13)
$ printf 'a\tb\tc\td\n' | ./detab 10 -15 +4
a         b   c   d
```

## Notes from the author

- The `-m +n` syntax is reminiscent of historical Unix tools (`tabs(1)` accepts similar). It's terse but unambiguous, and lets you specify both fixed and periodic stops in one invocation.
- The "next stop = min over candidates" pattern is the general way to combine multiple "what's my next stop" sources. Each source returns its best candidate; we pick the closest one.
- A more elaborate CLI would use `getopt_long` with `--stops=3,8,13` and `--every=4 --starting-at=16`. For a teaching exercise, the K&R syntax matches the era.
