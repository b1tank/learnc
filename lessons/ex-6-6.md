> Build a small `#define` preprocessor. It reads source lines; lines that start with `#define name value` register a macro; in all other lines, every occurrence of `name` is replaced by `value`. Other `#` directives can be ignored.

## Approach

Reuse the hash table from §6.6. Read input a line at a time. If a line starts with `#define`, parse `name` and the rest-of-line as `value`, and `install` them. Otherwise, scan the line for identifiers and substitute any that match a macro name.

```c:starter
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

#define HASHSIZE 101

struct nlist {
    struct nlist *next;
    char *name;
    char *defn;
};
static struct nlist *hashtab[HASHSIZE];

static unsigned hash(const char *s) {
    unsigned h = 0;
    while (*s) h = 31u * h + (unsigned char)*s++;
    return h % HASHSIZE;
}
static struct nlist *lookup(const char *name) {
    for (struct nlist *p = hashtab[hash(name)]; p; p = p->next)
        if (strcmp(name, p->name) == 0) return p;
    return NULL;
}
static void install(const char *name, const char *defn) {
    struct nlist *np = lookup(name);
    if (!np) {
        np = malloc(sizeof *np);
        np->name = strdup(name);
        unsigned h = hash(name);
        np->next = hashtab[h];
        hashtab[h] = np;
    } else free(np->defn);
    np->defn = strdup(defn);
}

static void substitute_line(const char *line) {
    const char *p = line;
    while (*p) {
        if (isalpha((unsigned char)*p) || *p == '_') {
            const char *start = p;
            while (isalnum((unsigned char)*p) || *p == '_') ++p;
            int len = (int)(p - start);
            char buf[256];
            if (len < (int)sizeof buf) {
                memcpy(buf, start, len);
                buf[len] = '\0';
                struct nlist *m = lookup(buf);
                if (m) fputs(m->defn, stdout);
                else   fwrite(start, 1, len, stdout);
            } else {
                fwrite(start, 1, len, stdout);
            }
        } else {
            putchar(*p++);
        }
    }
}

static void trim_newline(char *s) {
    size_t n = strlen(s);
    while (n > 0 && (s[n-1] == '\n' || s[n-1] == ' ' || s[n-1] == '\t'))
        s[--n] = '\0';
}

int main(void) {
    char line[1024];
    while (fgets(line, sizeof line, stdin)) {
        if (strncmp(line, "#define", 7) == 0) {
            const char *p = line + 7;
            while (*p == ' ' || *p == '\t') ++p;
            const char *name_start = p;
            while (isalnum((unsigned char)*p) || *p == '_') ++p;
            int name_len = (int)(p - name_start);
            while (*p == ' ' || *p == '\t') ++p;
            char name[128];
            if (name_len > 0 && name_len < (int)sizeof name) {
                memcpy(name, name_start, name_len); name[name_len] = '\0';
                char defn[1024];
                strncpy(defn, p, sizeof defn - 1);
                defn[sizeof defn - 1] = '\0';
                trim_newline(defn);
                install(name, defn);
            }
        } else if (line[0] == '#') {
            /* ignore other directives */
        } else {
            substitute_line(line);
        }
    }
    return 0;
}
```

Try it on:

```
#define MAX 100
#define HALF MAX/2
int x = MAX;
int y = HALF;
```

Output:

```
int x = 100;
int y = MAX/2;
```

Note: this is **single-pass non-recursive** substitution. `HALF` expanded once to `MAX/2`; we don't re-expand `MAX` in the result. Real cpp does iterative expansion with cycle detection — that's a much bigger project.

## What's missing vs. real cpp

- Function-like macros (`#define MAX(a,b) ...`).
- Recursive expansion (replace until no more macros match).
- `#if`/`#ifdef`/`#endif` conditionals.
- `#include` resolution.
- Stringification (`#x`) and concatenation (`a##b`).
- Predefined macros (`__LINE__`, `__FILE__`).

But the *structure* — a symbol table plus line-by-line substitution — is the heart of cpp.

## Try it

1. Make substitution recursive: after expanding once, re-scan and re-expand. Add a depth limit to catch infinite recursion.
2. Support `#undef name`.
3. Add function-like macros (parse `name(args)` after `#define`).

## Notes from the author

- The C preprocessor is its own little language — context-free grammar, hygiene rules, expansion semantics. Writing a "real" cpp takes thousands of lines. This 80-line stub captures the spirit.
- Replacing identifiers with stored text is the simplest form of macro expansion. Lisp-style macros do AST-level rewriting, which is cleaner but requires you to parse first; cpp's textual style is simpler but error-prone.
- The exercise teaches the **scanner-substitutor pattern** that recurs in templating engines, mail-merge programs, and shell variable expansion. Same skeleton, different keys and values.
