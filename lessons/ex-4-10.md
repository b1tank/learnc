> Refactor the calculator so it reads one *line* of input at a time (via something like our `getline` from §1.9), then tokenises that line. No `getch`/`ungetch` needed.

## Approach

Read a full line, then maintain an index into it. Tokenisation walks the index forward, skipping whitespace and grouping digits, operators, and identifiers. Pushback is implicit: if you decide a character isn't part of the current token, just don't advance the index.

```c:starter
#include <stdio.h>
#include <ctype.h>
#include <stdlib.h>
#include <string.h>

#define MAXLINE 200

/* read a line from stdin into s (up to lim-1 chars), return length */
int get_line(char s[], int lim) {
    int i = 0, c;
    while (i < lim - 1 && (c = getchar()) != EOF && c != '\n')
        s[i++] = (char)c;
    if (c == '\n') s[i++] = '\n';
    s[i] = '\0';
    return i;
}

/* token kinds */
enum { TOK_NUMBER, TOK_OPERATOR, TOK_END };

/* read one token; on number, fill *value; on operator, fill *op */
int next_token(const char line[], int *pos, double *value, char *op) {
    while (line[*pos] == ' ' || line[*pos] == '\t')
        ++*pos;
    if (line[*pos] == '\0' || line[*pos] == '\n')
        return TOK_END;

    if (isdigit((unsigned char)line[*pos]) ||
        (line[*pos] == '-' && isdigit((unsigned char)line[*pos + 1]))) {
        char buf[32];
        int  j = 0;
        if (line[*pos] == '-') buf[j++] = line[(*pos)++];
        while (isdigit((unsigned char)line[*pos]) || line[*pos] == '.')
            buf[j++] = line[(*pos)++];
        buf[j] = '\0';
        *value = atof(buf);
        return TOK_NUMBER;
    }

    *op = line[(*pos)++];
    return TOK_OPERATOR;
}

#define MAXVAL 100
int    sp = 0;
double val[MAXVAL];
void   push(double f) { if (sp < MAXVAL) val[sp++] = f; }
double pop(void)      { return (sp > 0) ? val[--sp] : 0.0; }

int main(void) {
    char line[MAXLINE];
    while (get_line(line, MAXLINE) > 0) {
        int pos = 0;
        double v;
        char   op;
        int    kind;
        while ((kind = next_token(line, &pos, &v, &op)) != TOK_END) {
            if (kind == TOK_NUMBER) {
                push(v);
            } else {
                double op2 = pop(), op1 = pop();
                switch (op) {
                    case '+': push(op1 + op2);              break;
                    case '-': push(op1 - op2);              break;
                    case '*': push(op1 * op2);              break;
                    case '/': push(op2 ? op1/op2 : 0.0);    break;
                }
            }
        }
        if (sp > 0)
            printf("= %g\n", val[sp-1]);
    }
    return 0;
}
```

```output
(awaits input — try '2 3 +' or '5 7 *')
```

## Why this is cleaner

- No global pushback state.
- `next_token` is a function of `(line, position)` — pure, easy to test in isolation.
- Look-ahead is just `line[pos + 1]`. Multi-character look-ahead is just reading ahead in the buffer.
- Error reporting can include "near column 17 of line `...`" — nice for users.

## Notes from the author

- "Read a line, then tokenise" is the standard parser scaffold. Lexers/parsers in real languages (Python's, Lua's, the JSON parser in your nearest library) all follow this shape: tokenise top-down, pass tokens to a parser, parser consumes them.
- Out-parameters (`int *pos, double *value, char *op`) are how C functions return multiple values. Modern C still does this; modern languages use tuples or destructuring. The mechanics are the same.
- For interactive REPLs, line-buffered input is *almost always* the right design. Character-by-character input only matters for raw-mode terminal programs (`vim`, `less`, `top`).
