> The `dcl` program from K&R §5.12 (a declaration-to-English translator) crashes or misbehaves on malformed input. Make it recover gracefully — print an error message and skip to the next declaration.

## Approach

Three categories of error in the original `dcl`:

1. **Missing identifier** — `int *();` (no name in the middle).
2. **Mismatched parentheses or brackets** — `int (*p[5);`.
3. **Unexpected tokens** — non-identifier where one was expected.

For each, set an error flag and bail out of the recursive descent. After printing an error message, consume the rest of the line (everything up to `\n`) so the next call starts fresh.

```c:starter
#include <stdio.h>
#include <ctype.h>
#include <string.h>

/* a minimal dcl skeleton with error recovery */

#define MAXTOKEN 100

enum { NAME, PARENS, BRACKETS, ERROR };

static char token[MAXTOKEN];
static int  tokentype;
static char name[MAXTOKEN];
static char datatype[MAXTOKEN];
static char out[1024];

static int err_flag = 0;

static int gettoken(void) {
    int c;
    char *p = token;
    while ((c = getchar()) == ' ' || c == '\t')
        ;
    if (c == '(') {
        if ((c = getchar()) == ')') {
            strcpy(token, "()");
            return tokentype = PARENS;
        } else {
            ungetc(c, stdin);
            return tokentype = '(';
        }
    }
    if (c == '[') {
        for (*p++ = c; (*p++ = getchar()) != ']'; )
            if (p[-1] == '\n' || p[-1] == EOF) {
                err_flag = 1; return tokentype = ERROR;
            }
        *p = '\0';
        return tokentype = BRACKETS;
    }
    if (isalpha(c)) {
        for (*p++ = c; isalnum(c = getchar()); )
            *p++ = c;
        *p = '\0';
        ungetc(c, stdin);
        return tokentype = NAME;
    }
    return tokentype = c;     /* * ( ) \n etc */
}

static void dcl(void);

static void dirdcl(void) {
    if (tokentype == '(') {
        dcl();
        if (tokentype != ')') {
            fprintf(stderr, "error: missing ')'\n");
            err_flag = 1;
            return;
        }
    } else if (tokentype == NAME) {
        strcpy(name, token);
    } else {
        fprintf(stderr, "error: expected name or '(name)'\n");
        err_flag = 1;
        return;
    }
    while (gettoken() == PARENS || tokentype == BRACKETS) {
        if (tokentype == PARENS) strcat(out, " function returning");
        else                     { strcat(out, " array"); strcat(out, token); strcat(out, " of"); }
    }
}

static void dcl(void) {
    int ns = 0;
    while (gettoken() == '*')
        ++ns;
    dirdcl();
    while (ns-- > 0)
        strcat(out, " pointer to");
}

static void discard_line(void) {
    int c;
    while ((c = getchar()) != '\n' && c != EOF)
        ;
}

int main(void) {
    while (gettoken() != EOF) {
        strcpy(datatype, token);
        out[0] = '\0';
        dcl();
        if (err_flag) {
            err_flag = 0;
            discard_line();
            continue;
        }
        if (tokentype != '\n')
            fprintf(stderr, "error: trailing junk\n");
        else
            printf("%s: %s%s\n", name, out, datatype);
    }
    return 0;
}
```

```output
(awaits input — try 'int *p' or 'int *();')
```

## Notes from the author

- The pattern "set an error flag, return up the recursion, discard input up to a sync point" is the standard parser-recovery idiom. Real C parsers (GCC, Clang) do something similar but more sophisticated (panic-mode recovery, error production rules).
- "Discard up to newline" is a simple sync point. More elaborate parsers discard up to `;` or `}` for richer grammars.
- Defensive parsers are hard to test exhaustively. Build a small corpus of "bad input" examples and run them through every change. Otherwise you'll regress recovery silently.
