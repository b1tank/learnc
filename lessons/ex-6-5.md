> Add an `undef(name)` function to the hash table from §6.6. It removes the entry, frees the memory, and handles "name not present" gracefully.

## Approach

A linked-list delete with a previous-pointer trick:

```c
struct nlist **pp = &hashtab[hash(name)];
while (*pp && strcmp((*pp)->name, name) != 0)
    pp = &(*pp)->next;
if (*pp) {
    struct nlist *gone = *pp;
    *pp = gone->next;     /* unlink */
    free(gone->name);
    free(gone->defn);
    free(gone);
    return 0;             /* success */
}
return -1;                 /* not found */
```

The double-pointer `pp` points to the slot we'd update — initially the bucket head, then the `next` of the previous node. When we find the match, `*pp = (*pp)->next` correctly handles both "first in chain" and "middle of chain".

```c:starter
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define HASHSIZE 101

struct nlist {
    struct nlist *next;
    char *name;
    char *defn;
};

static struct nlist *hashtab[HASHSIZE];

static unsigned hash(const char *s) {
    unsigned h = 0;
    while (*s) h = 31u * h + (unsigned char)*s++;
    return h % HASHSIZE;
}

static struct nlist *lookup(const char *name) {
    for (struct nlist *p = hashtab[hash(name)]; p; p = p->next)
        if (strcmp(name, p->name) == 0) return p;
    return NULL;
}

static struct nlist *install(const char *name, const char *defn) {
    struct nlist *np = lookup(name);
    if (!np) {
        np = malloc(sizeof *np);
        np->name = strdup(name);
        unsigned h = hash(name);
        np->next = hashtab[h];
        hashtab[h] = np;
    } else {
        free(np->defn);
    }
    np->defn = strdup(defn);
    return np;
}

static int undef(const char *name) {
    struct nlist **pp = &hashtab[hash(name)];
    while (*pp && strcmp((*pp)->name, name) != 0)
        pp = &(*pp)->next;
    if (!*pp) return -1;
    struct nlist *gone = *pp;
    *pp = gone->next;
    free(gone->name);
    free(gone->defn);
    free(gone);
    return 0;
}

int main(void) {
    install("PI",  "3.14");
    install("E",   "2.71");
    install("MAX", "100");

    printf("PI = %s\n",   lookup("PI")  ? lookup("PI")->defn  : "(nil)");
    printf("undef E -> %d\n", undef("E"));
    printf("E now = %s\n", lookup("E") ? lookup("E")->defn : "(nil)");
    printf("undef NONE -> %d\n", undef("NONE"));
    return 0;
}
```

```output
PI = 3.14
undef E -> 0
E now = (nil)
undef NONE -> -1
```

## The double-pointer trick

The "pointer-to-pointer" pattern eliminates the special case of "is this node the head of the list?" Without it:

```c
if (head->name matches) { free(head); head = head->next; }
else {
    prev = head;
    cur = head->next;
    while (cur && ...) { prev = cur; cur = cur->next; }
    if (cur) { prev->next = cur->next; free(cur); }
}
```

Two code paths, two bugs to make. With the double pointer, one code path. Linus Torvalds famously cited this as a "good taste" example.

## Try it

1. Implement `rename(old, new)` that calls `undef(old)` then `install(new, defn_of_old)`. Be careful: read `defn` before deleting.
2. Add a `clear_all()` that frees every entry in the table.
3. Walk the table after `undef` and verify no entry has a `next` pointing to freed memory.

## Notes from the author

- The double-pointer pattern (`T **pp`) is one of the highest-payoff techniques in C. Once you see it, you'll spot uses everywhere: list insertion, list removal, tree splicing.
- Always pair `install` with `undef` for any data you intend to manage dynamically. Memory leaks from "install once, never undef" are a common source of long-running-program bloat.
- The naming `undef` (vs `delete` or `remove`) follows the C preprocessor's `#undef` directive — since this table mimics `#define`/`#undef`, the symmetry is intentional.
