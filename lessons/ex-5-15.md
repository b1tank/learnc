> Add a `-f` ("fold") option to the line sorter that treats uppercase and lowercase letters as equivalent, so "Apple" and "apple" compare equal.

## Approach

A custom string comparator that compares characters after lowercasing each:

```c
int cmp_fold(const char *s, const char *t) {
    while (*s && *t) {
        int a = tolower((unsigned char)*s++);
        int b = tolower((unsigned char)*t++);
        if (a != b) return a - b;
    }
    return (unsigned char)*s - (unsigned char)*t;
}
```

When called by `qsort`'s comparator, the case-folding happens transparently.

```c:starter
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

static int fold = 0;
static int reverse = 0;

static int cmp_fold(const char *s, const char *t) {
    while (*s && *t) {
        int a = tolower((unsigned char)*s++);
        int b = tolower((unsigned char)*t++);
        if (a != b) return a - b;
    }
    return (unsigned char)*s - (unsigned char)*t;
}

static int cmp_lines(const void *a, const void *b) {
    const char *sa = *(const char *const *)a;
    const char *sb = *(const char *const *)b;
    int r = fold ? cmp_fold(sa, sb) : strcmp(sa, sb);
    return reverse ? -r : r;
}

#define MAXLINES 5000
static char *lines[MAXLINES];

int main(int argc, char *argv[]) {
    for (int i = 1; i < argc; ++i) {
        if (strcmp(argv[i], "-r") == 0) reverse = 1;
        if (strcmp(argv[i], "-f") == 0) fold    = 1;
    }
    int n = 0;
    char buf[4096];
    while (n < MAXLINES && fgets(buf, sizeof buf, stdin)) lines[n++] = strdup(buf);
    qsort(lines, n, sizeof lines[0], cmp_lines);
    for (int i = 0; i < n; ++i) fputs(lines[i], stdout);
    return 0;
}
```

Usage:

```bash
$ printf 'apple\nBanana\nCherry\nbanana\n' | ./sort
Banana          # uppercase sorts before lowercase in ASCII
Cherry
apple
banana
$ printf 'apple\nBanana\nCherry\nbanana\n' | ./sort -f
apple
Banana          # B == b under folding; original order preserved
banana
Cherry
```

## Notes from the author

- `tolower((unsigned char)*s)` is critical: passing a possibly-negative `char` directly to `tolower` is undefined behaviour. The `(unsigned char)` cast is a CERT-C requirement.
- The "preserve original order among equal keys" property (last sort result) is called **stability**. `qsort` is NOT guaranteed stable; if you need stability with case-folding, use `mergesort` (BSD/GNU extension) or sort by `(folded_key, original_index)`.
- "Case-fold" gets thorny in Unicode (German `ß` lowercases to `ss`; Turkish `İ` lowercases to `i̇`). For ASCII-only programs, `tolower` is enough. For real Unicode you'd want ICU's `u_strFoldCase`.
